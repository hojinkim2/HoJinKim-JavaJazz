public class LocalNumber {
    public static void main(String[] args) {
        int a = 0;
        double b;

        System.out.print(b); // 초기화 안하면 사용불가
        System.out.print(a + c); // c 변수는 아직 선언 안해서 사용 불가

        int c = 0;

        public double b = 0.0; // 지역변수는 public으로 지정 불가

        for (int e = 0; e < 10; e++) {
            int a = 1; // 3행에서 선언된 지역변수, 블록이 달라도 같은이름으로 다시 선언 불가
            System.out.print(e);
        }
    }
}
